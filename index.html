// Antigravity Inference Bridge
async function inferenceRequest(rawData) {
  const cleaned = cleanNullValues(rawData);
  const payload = formatToGemmaJSON(cleaned);
  const stream = await antiGravity.call({
    endpoint: "gemma-predict",
    data: payload,
    mode: "stream"
  });
  return stream;
}
